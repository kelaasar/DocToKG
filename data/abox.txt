    :are_for assertions:NASAsNuclearFlightSafetyActivities .

    :is_titled "Presidential Memorandum on Launch of Spacecraft Containing Space Nuclear Systems"@en ;
    :is_dated "August 20, 2019"^^xsd:string .

    :is_a_part_of assertions:BroaderNASAEmergencyManagementActivities .

    :contain "other radioactive material" .

assertions:TheFramework rdf:type :Framework ;
    :implements assertions:OtherRequirementsGuidanceAndProcesses .

assertions:OtherRequirementsGuidanceAndProcesses rdf:type :Process ;
    :are_relevant_to assertions:NuclearFlightSafety .
