    rdfs:domain :ProgramRequirement, :Impact ;
    rdfs:range :Date .

    rdfs:domain :Document ;
    rdfs:range :Public .

:is_authorized_by a owl:ObjectProperty ;
    rdfs:domain :Document ;
    rdfs:range :Law .

:is_incorporated_into a owl:ObjectProperty ;
    rdfs:domain :Document ;
    rdfs:range :Contract .

:is_uncontrolled_when a owl:ObjectProperty ;
    rdfs:domain :Document ;
    rdfs:range :Version .

:to_verify_that_this_is_the_correct_version_before_use a owl:ObjectProperty ;
